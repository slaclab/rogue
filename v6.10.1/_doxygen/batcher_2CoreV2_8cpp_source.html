149bool rpb::CoreV2::processFrame(ris::FramePtr frame) {
150 uint8_t temp;
151 uint32_t rem;
152 uint32_t fSize;
153 uint8_t dest;
154 uint8_t fUser;
155 uint8_t lUser;
156
157 // Reset old data
158 reset();
159
160 ris::FrameIterator beg;
161 ris::FrameIterator mark;
162 ris::FrameIterator tail;
163
164 // Drop errored frames
165 if ((frame->getError())) {
166 log_->warning("Dropping frame due to error: 0x%" PRIx8, frame->getError());
167 return false;
168 }
169
170 // Drop small frames: 10 = 2 byte header + 1 byte payload + 7 byte tail
171 if ((rem = frame->getPayload()) < 10) {
172 log_->warning("Dropping small frame size = %" PRIu32, frame->getPayload());
173 return false;
174 }
175
176 // Get version & size
177 beg = frame->begin();
178 ris::fromFrame(beg, 1, &temp);
179
181 // Super-Frame Header in firmware: 2 bytes
183 // v.txMaster.tValid := '1';
184 // v.txMaster.tData(3 downto 0) := x"2"; -- Version = 0x2
185 // v.txMaster.tData(7 downto 4) := Reserved
186 // v.txMaster.tData(15 downto 8) := r.seqCnt;
187 // ssiSetUserSof(AXIS_CONFIG_G, v.txMaster, '1');
189
190 // Check version, convert width
191 if ((temp & 0xF) != 2) {
192 log_->error("Version mismatch. Got %" PRIu8, (temp & 0xF));
193 return false;
194 }
195
196 // Get sequence #
197 ris::fromFrame(beg, 1, &seq_);
198
199 // Store frame reference now that validation has passed
200 frame_ = frame;
201
202 // Compute remaining frame size
203 rem -= headerSize_;
204
205 // Set marker to end of frame
206 mark = frame->end();
207
208 // Process each frame, stop when we have reached just after the header
209 while (mark != beg) {
210 // sanity check
211 if (rem < tailSize_) {
212 log_->error("Not enough space (%" PRIu32 ") for tail (%" PRIu32 ")", rem, tailSize_);
213 reset();
214 return false;
215 }
216
217 // Jump to start of the tail
218 mark -= tailSize_;
219 rem -= tailSize_;
220
221 // Add tail iterator to end of list
222 tails_.push_back(mark);
223
224 // Get tail data, use a new iterator
225 tail = mark;
226 ris::fromFrame(tail, 4, &fSize);
227 ris::fromFrame(tail, 1, &dest);
228 ris::fromFrame(tail, 1, &fUser);
229 ris::fromFrame(tail, 1, &lUser);
230
231 // Not enough data for rewind value
232 if (fSize > rem) {
233 log_->error("Not enough space (%" PRIu32 ") for frame (%" PRIu32 ")", rem, fSize);
234 reset();
235 return false;
236 }
237
238 // Set marker to start of data
239 mark -= fSize;
240 rem -= fSize;
241
242 // Create data record and add it to end of list
243 list_.push_back(rpb::Data::create(mark, fSize, dest, fUser, lUser));
244 }
245 return true;
246}
