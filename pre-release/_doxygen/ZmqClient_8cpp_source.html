40constexpr uint32_t kRetryLogIntervalSeconds = 30;
41
42void logWaitRetry(rogue::LoggingPtr log, double seconds, uint32_t& lastLoggedSeconds) {
43 uint32_t elapsedSeconds = static_cast<uint32_t>(seconds);
44
45 if (elapsedSeconds == 0) return;
46
47 if (lastLoggedSeconds == 0) {
48 log->warning(
49 "Timeout waiting for response after %" PRIu32 " seconds, server may be busy. Continuing to wait...",
50 elapsedSeconds);
51 lastLoggedSeconds = elapsedSeconds;
52 return;
53 }
54
55 if (elapsedSeconds >= (lastLoggedSeconds + kRetryLogIntervalSeconds)) {
56 log->warning("Still waiting for response after %" PRIu32 " seconds, server may be busy...", elapsedSeconds);
57 lastLoggedSeconds = elapsedSeconds;
58 }
59}
60
61void logWaitRecovered(rogue::LoggingPtr log, double seconds) {
62 uint32_t elapsedSeconds = static_cast<uint32_t>(seconds);
63
64 if (elapsedSeconds != 0) {
65 log->warning("Received response from server after %" PRIu32 " seconds.", elapsedSeconds);
66 }
67}

204std::string rogue::interfaces::ZmqClient::sendString(const std::string& path, const std::string& attr, const std::string& arg) {
205 std::string snd;
206 std::string ret;
207 zmq_msg_t msg;
208 std::string data;
209 double seconds = 0;
210 uint32_t lastLoggedSeconds = 0;
211
212 if (!doString_) throw rogue::GeneralError::create("ZmqClient::sendString", "Invalid send call in standard mode");
213
214 snd = "{\"attr\": \"" + attr + "\",";
215 snd += "\"path\": \"" + path + "\"";
216
217 if (arg != "") snd += ",\"args\": [\"" + arg + "\"]";
218
219 snd += "}";
220
221 rogue::GilRelease noGil;
222 std::lock_guard<std::mutex> lock(reqLock_);
223 zmq_send(this->zmqReq_, snd.c_str(), snd.size(), 0);
224
225 while (1) {
226 zmq_msg_init(&msg);
227 if (zmq_recvmsg(this->zmqReq_, &msg, 0) <= 0) {
228 seconds += static_cast<double>(timeout_) / 1000.0;
229 if (waitRetry_) {
230 logWaitRetry(log_, seconds, lastLoggedSeconds);
231 zmq_msg_close(&msg);

233 zmq_msg_close(&msg);
234 throw rogue::GeneralError::create("ZmqClient::sendString",
235 "Timeout waiting for response after %d Seconds.",
236 static_cast<int>(seconds));
237 }
238 } else {
239 break;
240 }
241 }
242
243 logWaitRecovered(log_, seconds);
244
245 data = std::string((const char*)zmq_msg_data(&msg), zmq_msg_size(&msg));
246 zmq_msg_close(&msg);

268bp::object rogue::interfaces::ZmqClient::send(bp::object value) {
269 zmq_msg_t txMsg;
270 zmq_msg_t rxMsg;
271 Py_buffer valueBuf;
272 bp::object ret;
273 double seconds = 0;
274 uint32_t lastLoggedSeconds = 0;
275
276 if (doString_) throw rogue::GeneralError::create("ZmqClient::send", "Invalid send call in string mode");
277
278 if (PyObject_GetBuffer(value.ptr(), &(valueBuf), PyBUF_SIMPLE) < 0)
279 throw(rogue::GeneralError::create("ZmqClient::send", "Failed to extract object data"));
280
281 zmq_msg_init_size(&txMsg, valueBuf.len);
282 memcpy(zmq_msg_data(&txMsg), valueBuf.buf, valueBuf.len);
283 PyBuffer_Release(&valueBuf);
284
285 {
286 rogue::GilRelease noGil;
287 std::lock_guard<std::mutex> lock(reqLock_);
288 zmq_sendmsg(this->zmqReq_, &txMsg, 0);
289
290 while (1) {
291 zmq_msg_init(&rxMsg);
292 if (zmq_recvmsg(this->zmqReq_, &rxMsg, 0) <= 0) {
293 seconds += static_cast<double>(timeout_) / 1000.0;
294 if (waitRetry_) {
295 logWaitRetry(log_, seconds, lastLoggedSeconds);
296 zmq_msg_close(&rxMsg);
297 } else {
298 zmq_msg_close(&rxMsg);
299 throw rogue::GeneralError::create(
300 "ZmqClient::send",
301 "Timeout waiting for response after %d Seconds, server may be busy!",
302 static_cast<int>(seconds));
303 }
304 } else {
305 break;
306 }
307 }
308 }
309
310 logWaitRecovered(log_, seconds);
311
312 PyObject* val = Py_BuildValue("y#", zmq_msg_data(&rxMsg), zmq_msg_size(&rxMsg));
313
314 if (val == NULL) throw(rogue::GeneralError::create("ZmqClient::send", "Failed to generate bytearray"));
315
316 zmq_msg_close(&rxMsg);
317
318 bp::handle<> handle(val);
319 ret = bp::object(handle);
320 return ret;
321}
