147bool rim::TcpClient::waitReady(double timeout, double period) {
148 uint32_t id;
149 uint64_t addr = 0;
150 uint32_t size = 0;
151 uint32_t type = rim::TcpBridgeProbe;
152 zmq_msg_t msg[4];
153 auto deadline = std::chrono::steady_clock::now() + std::chrono::duration<double>(timeout);
154
155 // Retry the probe until the overall timeout expires. This verifies the
156 // full request/response bridge path, not just local socket setup.
157 while (std::chrono::steady_clock::now() < deadline) {
158 {
159 std::lock_guard<std::mutex> probeLock(probeMtx_);
160 // Track the current probe ID so the response thread can match the
161 // incoming bridge reply back to this wait operation.
162 probeId_ = ++probeSeq_;
163 probeDone_ = false;
164 probeResult_.clear();
165 id = probeId_;
166 }
167
168 // Build a minimal internal bridge-control request. The server handles
169 // this locally and responds with the normal six-part completion
170 // message, but without touching downstream memory.
171 zmq_msg_init_size(&(msg[0]), 4);
172 std::memcpy(zmq_msg_data(&(msg[0])), &id, 4);
173 zmq_msg_init_size(&(msg[1]), 8);
174 std::memcpy(zmq_msg_data(&(msg[1])), &addr, 8);
175 zmq_msg_init_size(&(msg[2]), 4);
176 std::memcpy(zmq_msg_data(&(msg[2])), &size, 4);
177 zmq_msg_init_size(&(msg[3]), 4);
178 std::memcpy(zmq_msg_data(&(msg[3])), &type, 4);
179
180 {
181 std::lock_guard<std::mutex> block(bridgeMtx_);
182 // Send through the same request socket and multipart framing used
183 // by normal memory transactions so this exercises the real bridge
184 // path end-to-end.
185 for (uint32_t x = 0; x < 4; ++x) {
186 if (zmq_sendmsg(this->zmqReq_, &(msg[x]), (x == 3 ? 0 : ZMQ_SNDMORE) | ZMQ_DONTWAIT) < 0) {
187 bridgeLog_->debug("Readiness probe send failed for port %s", this->reqAddr_.c_str());
188 break;
189 }
190 }
191 }
192
193 for (uint32_t x = 0; x < 4; ++x) zmq_msg_close(&(msg[x]));
194
195 std::unique_lock<std::mutex> probeLock(probeMtx_);
196 // The receive thread sets probeDone_/probeResult_ when it sees a probe
197 // response with the matching ID. We only wait for one retry period
198 // here so we can re-send if the server is not up yet.
199 probeCond_.wait_for(probeLock, std::chrono::duration<double>(period), [&]() { return probeDone_ && probeId_ == id; });
200
201 if (probeDone_ && probeId_ == id && probeResult_ == "OK") {
202 bridgeLog_->debug("Readiness probe succeeded for port %s", this->reqAddr_.c_str());
203 return true;
204 }
205 }
206
207 bridgeLog_->warning("Timed out waiting for bridge readiness on port %s", this->reqAddr_.c_str());
208 return false;
209}

228 uint32_t size;
229 uint32_t type;

239

244

249
250 // Type message

252 zmq_msg_init_size(&(msg[3]), 4);
253 std::memcpy(zmq_msg_data(&(msg[3])), &type, 4);
254
255 // Write transaction
256 if (type == rim::Write || type == rim::Post) {
257 msgCnt = 5;
258 zmq_msg_init_size(&(msg[4]), size);
259 std::memcpy(zmq_msg_data(&(msg[4])), tran->begin(), size);
260
261 // Read transaction
262 } else {
263 msgCnt = 4;
264 }
265
266 bridgeLog_->debug("Requested transaction id=%" PRIu32 ", addr=0x%" PRIx64 ", size=%" PRIu32 ", type=%" PRIu32
267 ", cnt=%" PRIu32 ", port: %s",
268 id,
269 addr,
270 size,
271 type,
272 msgCnt,
273 this->reqAddr_.c_str());
274
275 // Add transaction
276 if (type == rim::Post)
277 tran->done();
278 else
279 addTransaction(tran);
280
281 // Send message
282 for (x = 0; x < msgCnt; x++) {
283 if (zmq_sendmsg(this->zmqReq_, &(msg[x]), ((x == (msgCnt - 1) ? 0 : ZMQ_SNDMORE)) | ZMQ_DONTWAIT) < 0) {
284 bridgeLog_->warning("Failed to send transaction %" PRIu32 ", msg %" PRIu32 " on %s: %s",
285 id,

294void rim::TcpClient::runThread() {
295 rim::TransactionPtr tran;
296 bool err;
297 uint64_t more;
298 size_t moreSize;
299 uint32_t x;
300 uint32_t msgCnt;
301 zmq_msg_t msg[6];
302 uint32_t id;
303 uint64_t addr;
304 uint32_t size;
305 uint32_t type;
306 char result[1000];
307
308 bridgeLog_->logThreadId();
309
310 while (threadEn_) {
311 for (x = 0; x < 6; x++) zmq_msg_init(&(msg[x]));
312 msgCnt = 0;
313 x = 0;
314
315 // Get message
316 do {
317 // Get the message
318 if (zmq_recvmsg(this->zmqResp_, &(msg[x]), 0) >= 0) {
319 if (x != 5) x++;
320 msgCnt++;
321
322 // Is there more data?
323 more = 0;
324 moreSize = 8;
325 zmq_getsockopt(this->zmqResp_, ZMQ_RCVMORE, &more, &moreSize);
326 } else {
327 more = 1;
328 }
329 } while (threadEn_ && more);
330
331 // Proper message received
332 if (threadEn_ && (msgCnt == 6)) {
333 // Check sizes
334 if ((zmq_msg_size(&(msg[0])) != 4) || (zmq_msg_size(&(msg[1])) != 8) || (zmq_msg_size(&(msg[2])) != 4) ||
335 (zmq_msg_size(&(msg[3])) != 4) || (zmq_msg_size(&(msg[5])) > 999)) {
336 bridgeLog_->warning(
337 "Bad message sizes. id=%zu addr=%zu size=%zu type=%zu result=%zu",
338 zmq_msg_size(&(msg[0])),
339 zmq_msg_size(&(msg[1])),
340 zmq_msg_size(&(msg[2])),
341 zmq_msg_size(&(msg[3])),
342 zmq_msg_size(&(msg[5])));
343 for (x = 0; x < msgCnt; x++) zmq_msg_close(&(msg[x]));
344 continue; // while (1)
345 }
346
347 // Get return fields
348 std::memcpy(&id, zmq_msg_data(&(msg[0])), 4);
349 std::memcpy(&addr, zmq_msg_data(&(msg[1])), 8);
350 std::memcpy(&size, zmq_msg_data(&(msg[2])), 4);
351 std::memcpy(&type, zmq_msg_data(&(msg[3])), 4);
352
353 memset(result, 0, 1000);
354 std::strncpy(result, reinterpret_cast<char*>(zmq_msg_data(&(msg[5]))), zmq_msg_size(&(msg[5])));
355
356 if (type == rim::TcpBridgeProbe) {
357 std::lock_guard<std::mutex> probeLock(probeMtx_);
358 if (id == probeId_) {
359 probeResult_ = result;
360 probeDone_ = true;
361 probeCond_.notify_all();
362 }
363 for (x = 0; x < msgCnt; x++) zmq_msg_close(&(msg[x]));
364 continue; // while (1)
365 }
366
367 // Find Transaction
368 if ((tran = getTransaction(id)) == NULL) {
369 bridgeLog_->warning("Failed to find transaction id=%" PRIu32, id);
370 for (x = 0; x < msgCnt; x++) zmq_msg_close(&(msg[x]));
371 continue; // while (1)
372 }
373
374 // Lock transaction
375 rim::TransactionLockPtr lock = tran->lock();
376
377 // Transaction expired
378 if (tran->expired()) {
379 bridgeLog_->warning("Transaction expired. Id=%" PRIu32, id);
380 for (x = 0; x < msgCnt; x++) zmq_msg_close(&(msg[x]));
381 continue; // while (1)
382 }
383
384 // Double check transaction
385 if ((addr != tran->address()) || (size != tran->size()) || (type != tran->type())) {
386 bridgeLog_->warning("Transaction data mismatch. Id=%" PRIu32, id);
387 tran->error("Transaction data mismatch in TcpClient");
388 for (x = 0; x < msgCnt; x++) zmq_msg_close(&(msg[x]));
389 continue; // while (1)
390 }
391
392 // Copy data if read
393 if (type != rim::Write) {
394 if (zmq_msg_size(&(msg[4])) != size) {
395 bridgeLog_->warning("Transaction size mismatch. Id=%" PRIu32, id);
396 tran->error("Received transaction response did not match header size");
397 for (x = 0; x < msgCnt; x++) zmq_msg_close(&(msg[x]));
398 continue; // while (1)
399 }
400 std::memcpy(tran->begin(), zmq_msg_data(&(msg[4])), size);
401 }
402 if (strcmp(result, "OK") != 0)
403 tran->error(result);
404 else
405 tran->done();
406 bridgeLog_->debug("Response for transaction id=%" PRIu32 ", addr=0x%" PRIx64 ", size=%" PRIu32
407 ", type=%" PRIu32 ", cnt=%" PRIu32 ", port: %s, Result: (%s)",
408 id,
409 addr,
410 size,
411 type,
412 msgCnt,
413 this->respAddr_.c_str(),
414 result);
415 }
416 for (x = 0; x < msgCnt; x++) zmq_msg_close(&(msg[x]));
417 }
